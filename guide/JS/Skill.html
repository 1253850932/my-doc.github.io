                                                                      # 1. 字符串转对象

                                                                      let str = 'id=10&name=小明&age=18' 转化成 对象

                                                                      1. 字符串按&拆分，返回一个数组 [ 'id=10', 'name=小明', 'age=18' ]

                                                                      2. 遍历数组，再把每个字符串用 = 拆分

                                                                      id=10,name=小明,age=18-->[ 'id', '10' ],[ 'name', '小明' ],[ 'age', '18' ]

                                                                      1. 下班 0 作为空对象的 key 值

                                                                      2. 下表 1 作为空对象的 value 值 5. 遍历结束后，即可转换成功

                                                                      let str = 'id=10&name=小明&age=18'
let obj = {}
str.split('&').forEach(item => {
  let arr = item.split('=')
  obj[arr[0]] = arr[1]
})

                                                                      防抖

                                                                      function debounce(fn, delay) {
  let timer = 0
  return function () {
    if (timer) return
    let that = this
    let args = arguments
    timer = setTimeout(() => {
      fn.apply(that, args)
      timer = 0
    }, delay)
  }
}

                                                                      function throttle(fn, delay) {
  let timer = 0
  return function () {
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      fn.apply(that, args)
      timer = 0
    }, delay)
  }
}

                                                                      # 数组转为对象

                                                                      var menuArr = [
    [1, "Area1", -1],
    [2, "Area2", -1],
    [3, "Area1-1", 1],
    [4, "Area1-2", 1],
    [5, "Area2-1", 2],
    [6, "Area2-2", 2],
    [7, "Area1-2-3", 4],
    [8, "Area2-2-1", 6],
];

function arrToObj(arr) {
  // 将数组转换为对象
  var obj = arr.reduce (prev, cur) {
    // 解构出需要的元素
    var [id, name, parent_id] = cur;
    // 创建图元素对象
    var item = { id: id, name: name, parent_id: parent_id };

    // 判断该图元素是否是根节点
    if (parent_id === -1) {
      // 如果是根节点，则在它的对象中添加 children 属性，并将该对象添加到结果对象 prev 中
      item.children = {};
 prev[id] = item;
    } else {
      // 如果不是根节点，则在它的父节点对象中添加该子节点对象，并在该子节点对象中添加图元素转换后的对象
      if (prev[parent_id]) {
        prev[parent_id].children = prev[parent_id].children || {};
        prev[parent_id].children[id] = item;
      }
    }

    return prev; // 返回结果对象
  }, {});

  return obj; // 返回转换后的对象
}

                                                                      function DataType(tgt, type) {
  const dataType = Object.prototype.toString
    .call(tgt)
    .replace(/\[object (\w+)\]/, '$1')
    .toLowerCase()
  return type ? dataType === type : dataType
}

                                                                      1. Lazy loading functions 延迟加载函数

                                                                        当函数中的复杂判断分支越来越多时，可以大大节省资源开销，可以使用闭包来实现。

                                                                      function Func() {
  if (a === b) {
    console.log('x')
  } else {
    console.log('y')
  }
}

// replace with
function Func() {
  if (a === b) {
    Func = function () {
      console.log('x')
    }
  } else {
    Func = function () {
      console.log('y')
    }
  }
  return Func()
}

                                                                      2. Handle Async/Await parameters gracefully 优雅地处理 Async/Await 参数

                                                                      function AsyncTo(promise) {
  return promise.then(data => [null, data]).catch(err => [err])
}

const [err, res] = await AsyncTo(Func())

                                                                      function AutoResponse(width = 750) {
  const target = document.documentElement
  target.clientWidth >= 600
    ? (target.style.fontSize = '80px')
    : (target.style.fontSize = (target.clientWidth / width) * 100 + 'px')
}

<p>照片区域：</p>

                                                                      window.onload = function () {
  video = document.querySelector('video')
  canvas = document.querySelector('canvas')
  ctx = canvas.getContext('2d')

  webCamera()
  // 申请网络摄像头操作权限
  function webCamera() {
    //检查浏览器是否支持getUserMedia方法
    navigator.getUserMedia =
      navigator.getUserMedia ||
      navigator.webkitGetUserMedia ||
      navigator.mozGetUserMedia
    if (navigator.getUserMedia) {
      navigator.getUserMedia(
        {
          //关闭了音频
          audio: false,
          //指定了视频的宽度和高度
          video: {
            width: 300,
            height: 200,
          },
        },
        //如果成功获取到视频流
        function (stream) {
          //视频流，则将其赋值给video元素的srcObject属性
          video.srcObject = stream
          // 视频元数据加载完成后播放视频
          video.onloadedmetadata = function (e) {
            video.play()
          }
        },
        function (err) {
          console.log('Error occured:' + err.name)
        }
      )
    } else {
      console.log("this navigator doesn't support webcam!")
    }
  }

  // 保存图片
  function takePhoto() {
    ctx.drawImage(video, 0, 0, 300, 200)
  }
}
